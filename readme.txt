# BUILD plugin
    sudo ./tsxs -o add_text.so -c ~/add_text.cc

# INSTALL plugin
    sudo ./tsxs -o add_text.so -i

# CONFIGURE plugin (/opt/ts//etc/trafficserver/plugin.config)
    add_text.so Created with NetCDN Origin Packager
